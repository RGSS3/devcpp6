#include "LangFile.h"

LangFile::LangFile() {
}

LangFile::LangFile(const char* FileName) {
	OpenFile(FileName);
}

LangFile::~LangFile() {
}

void LangFile::OpenFile(const char* FileName) {
	FILE* file = fopen(FileName,"rb");
	if(file) {
		char linebuffer[512];
		char textbuffer[512];
		int ID;
		while(fgets(linebuffer,512,file)) {
			if(sscanf(linebuffer,"%d=%[^\r\n]",&ID,textbuffer) == 2) {
				LangItem item;
				item.ID = ID;
				item.text = textbuffer;

				// Find formatting specifiers
				// Loop through format options
				int offset = 0;
				int pos = item->text.find('%',offset);
				if(pos != string::npos) {

					contents.push_back(item);
				}
			}
		}

		Log("Found %d items in file %s\r\n",contents.size(),FileName);

		fclose(file);
	}
}

LangItem* LangFile::FindID(int ID) {
	for(unsigned int i = 0; i < contents.size(); i++) {
		if(contents[i].ID == ID) {
			return &contents[i];
		}
	}
	return NULL;
}
